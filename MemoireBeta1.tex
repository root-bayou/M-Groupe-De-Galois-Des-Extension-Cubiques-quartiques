\chapter{EXTENSION DE CORPS}
\lhead{CHAPITRE 1}
\rhead{EXTENSION DE CORPS }
Dans tout ce qui suit $K$ désigne un corps commutatif.

\section{Propriété élémentaires des extensions de corps}
	
\begin{definition}
On appelle extension de $K$, la donnée d'un corps $L$ et d'un homomorphisme
$i:\,K\longrightarrow L$ de telle sorte qu'on ait l'isomorphisme $K\simeq i\left(K\right)$.\\
L'homomorphisme $i$  est appelé plongement de $K$ dans $L$.
\end{definition}
%
Ainsi définie, $K$ est un sous-corps de $L$, à isomorphisme près.\\
Lorsque $K\subset L$, le corps $L$ est alors considéré comme un $K$-espace vectoriel. Soit,
dans ce cas, $\alpha\in L$  et définissons le morphisme d'anneaux suivant
\[
\begin{array}{cccc}
\phi_{\alpha}: & K\left[X\right] & \longrightarrow & K\left[\alpha\right]\\
 & X & \longrightarrow & \alpha
\end{array}
\]
 de noyau $\ker\phi_{\alpha}$ qui est un idéal premier et principal. 
\begin{definition}
On dit que l'élément $\alpha\in L$ est algébrique lorsque $\ker\phi_{\alpha}\ne\left\{ 0\right\} $,
sinon il sera dit transcendant. 
\end{definition}
\begin{flushleft}
		\textbf{Notation :} L'extension $L$ d'un corps $K$ est notée $\, L /K $.
\end{flushleft}
\begin{example}
\begin{enumerate}
    \item Tout corps de caractéristique 0 est extension du corps $\,\mathbb{Q}$.
   \item  Tout corps de caractéristique $p \ne 0$ est extension du corps $\,\mathbb{Z}\, /\, p\mathbb{Z}\,$.\end{enumerate}
\end{example}
	%
	\begin{definition}
		On dit qu'un corps $M$ est un \textbf{corps intermédiaire} pour une extension $L /K $, si $K \subset M \subset L$.
	\end{definition}
	%
\begin{definition}
		Soit $n \in \mathbb{N}^*$. \\
		Le corps $K(\alpha_1,\alpha_2,\ldots ,\alpha_n)$, où les $\alpha_i$, $1 \leq\alpha_i\leq n $ sont des élément de $L$ , est appelée extension de $K$ obtenue par adjonctions successives de ces éléments. On écrira alors :
\begin{equation*}
\begin{split}
				K(\alpha_1,\alpha_2) = K(\alpha_1)(\alpha_2) \,, K(\alpha_1,\alpha_2 ,\alpha_3) =  K(\alpha_1,\alpha_2)(\alpha_3),\ldots\\
				\ldots ,K(\alpha_1,\alpha_2,\ldots ,\alpha_n) =
				K(\alpha_1,\alpha_2,\ldots ,\alpha_{n-1})(\alpha_n).
			\end{split}
\end{equation*}
\end{definition}
%
\begin{definition}
		Soit $ L/K $ une extension de corps. On suppose que $K \subseteq L$; alors pour tout partie non vide $\delta$ de $L$, le sous-corps de $L$ engendré par
		$K \cup \delta$ est noté $K(\delta)$, appelé extension de $K$ obtenue par l'adjonction de $\delta$ à $K$.
	\end{definition}
	\begin{remark}
		Pour $\delta = \{\alpha\}$, où  $\alpha \in L$, $K\left(\delta\right)$ s'écrit $K\left(\alpha\right)$, elle est dite \textbf{extension simple} de $K$ obtenue par l'adjonction de $\alpha \,\text{à }\, K$. \\
	\end{remark}

%
		
	\subsection{degré d'une extension de corps}
	Pour une extension $L/K$ donnée, le corps $L$ est considéré comme un espace vectoriel sur $K$.
	Rappelons que tout espace vectoriel  de dimension  finie admet une base finie et toutes les bases ont le même cardinale, appelé dimension de cet espace vectoriel. 
	
	\begin{definition} Soit $L/K$ une extension de corps.\\
	On appelle degré de l'extension $L/K$, noté $[L/K]$,  la dimension du $K$-espace vectoriel $L$.\\
	Lorsque la dimension du  $K$-espace vectoriel $L$ est  finie, l'extension $L/K$ est dite de degré finie et on note par $[L/K]=\dim_K L$ le degré de cette extension.
	\end{definition}
	%
	\begin{theorem}{(Théorème de la base télescopique)}\label{th1}
		Étant données les extensions de corps $L / K$ et $M / L$, alors
		\begin{equation}\label{eq2}
			[M/K] = [M / L]\times[L / K].
		\end{equation}
	\end{theorem}
	\begin{proof}
		(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [6]}).
	\end{proof}	
		Du théorème \eqref{th1}, on déduit le corolaire suivant.
	\begin{corollary}
		soit $L / K$ une extension de corps, $[L / K] < \infty$; si,$ \:\left\lbrace K_i\right\rbrace $,\\
		$r \in \mathbb{N}^{*} \,, 1 < r \leq n $ est une famille finie, totalement ordonnée, de corps intermédiaires, alors
		\begin{equation*}
			\begin{split}
				K \subseteq K_1 \subseteq K_2 \subseteq \ldots \subseteq K_r \subseteq L \; implique \\
				[L / K] = [L / K_r][K_r / K_{r-1}] \ldots [K_1 / K].
			\end{split}
		\end{equation*}

	\end{corollary}
\section{Isomorphisme d'extension de corps }
	Une extension $L$ de $K$ est définie par la donnée d'un couple $(L,i)$, où $i$ est un plongement de $K$ dans $L$.
	\begin{definition}
		Soit $K$ et $L$ deux corps isomorphes, on dit que les extensions $L/K$ et $M / F$, définies respectivement par les couple $(L,i)$ et $(M,j)$ sont isomorphes, s'il existe un couple d'isomorphisme $(\lambda,\mu)$, respectivement, de $K$ sur $F$ et de $L$ sur $M$, tel que
		le diagramme suivant est commutatif :
		\begin{equation*}
			\xymatrix{
				K \ar[r]^\lambda \ar[d]^i  & F \ar[d]^j \\
				L \ar[r]^\mu & M
			}
		\end{equation*}
		C'est à  dire $\mu \circ i = j \circ \lambda $. \\
		On dit alors que le couple $(\lambda,\mu)$ est un isomorphismes d'extensions de corps de $L/K$ sur $M / F$. \\

		En particulier :
		\begin{itemize}
			\item[]{$\bullet$} Si $K \subseteq L$ et $F \subseteq M$, on a,
			\begin{equation*}
				\mu \circ i = j \circ \lambda \iff \mu_{/K} = \lambda.
			\end{equation*}
			\item[]{$\bullet$} Si $K = F \,, \lambda = id_k$, le diagramme devient alors,
		\end{itemize}
		\begin{equation*}
			\xymatrix{
				K \ar[d]^i \ar[rd]^j\\
				L \ar[r]^\mu & M
			}
		\end{equation*}
		D'où $\mu \circ i = j$. \\

		Si de plus, $\mu \,, \alpha$ sont les injections canoniques alors,
		\begin{equation*}
			\mu \circ i = j  \iff \mu_{/K} = id_K.
		\end{equation*}
		et dans ce cas on dit que les corps $M$ et $L$ sont conjugués sur $K$, (on dit aussi que ils sont $K$-isomorphes).
	\end{definition}

	\section{Extensions algébriques - Clôtures algébriques}
	\subsection{\'Eléments algébriques}
	\begin{definition}
		Soit $L/K$ une extension de corps, soit $\alpha \in L$ .
		$\alpha$ est dit algébrique sur $K$, s'il existe un polynôme $p(X)\in K[X]$, non nul,  tel que $p(\alpha) = 0 $. Dans ce cas, l'extension $K(\alpha)$ est dite \textbf{simple et algébrique} sur $K$.\\
		Lorsque $\alpha$  n'est pas algébrique il est dit transcendant
	\end{definition}
	%
	\begin{proposition}
		Soit $L /K$ une extension de corps, pour tout $\alpha$ dans $L$ on associe l'application
		\begin{equation*}
			\begin{split}
				f_\alpha : K [X] \longrightarrow L \\
				g(X) \longrightarrow g(\alpha)
			\end{split}
		\end{equation*}
		où $f_\alpha$ désigne un morphisme d'anneaux unitaires et on a
		\begin{equation*}
			f_\alpha \, est \, non \, injectif \iff \alpha \, est \, algébrique \, sur \, K
		\end{equation*}

	\end{proposition}
	\begin{proof} $K[\alpha]$   étant un sous-anneau du corps $L$, il est donc intègre et   $K[\alpha] = Imf_\alpha$. De plus, on a les équivalences suivantes
		\begin{equation*}
			\begin{array}{r c l}
				f_\alpha \, est \, non \, injectif & \iff & Kerf_\alpha \ne \left\lbrace 0\right\rbrace \\
				& \iff & \exists g(X) \in K[X]\textbackslash \left\lbrace 0 \right\rbrace \, tel \, que \, g(\alpha) = 0 \\
				& \iff & \alpha \, est \, algébrique \, sur \, K
			\end{array}
		\end{equation*}
	\end{proof}
%
	\begin{definition}
		Soit $L/K$ une extension de corps, $\alpha \in L$ un élément algébrique sur $K$, le polynôme non constant $p(X)$, unitaire et irréductible de $K[X]$, vérifiant $p(\alpha) = 0$, est appelé le polynôme minimal de $\alpha$ sur $K$, noté $irr_K(\alpha,X)$.
	\end{definition}
	Le résultat ci-dessous donne une caractérisation du polynôme minimal d'un élément algébrique.
	\begin{theorem}
		Soit $L /K$ une extension de corps, si $\alpha \in L$ et un élément algébrique sur $K$, alors
		\begin{enumerate}
			\item Il existe un unique polynôme $p(X)$, unitaire et irréductible de $K[X]$ vérifiant
			\begin{equation}\label{t1}
				\forall f(X) \in K[X]\textbackslash\left\lbrace 0\right\rbrace \, et \, f(\alpha) = 0 \iff p(X)|f(X) \, dans \, K[X].
			\end{equation}
			\item L'extension simple $K(\alpha) /K$ vérifie
			\begin{equation}\label{t2}
				K(\alpha)=\left\lbrace f(\alpha), \, f(X) \in K[X]\right\rbrace .
			\end{equation}
			\item \begin{equation}\label{t3}
				[L / K] = [K(\alpha) / K] = deg \,p .
			\end{equation}
		\end{enumerate}
	\end{theorem}
	\begin{proof} (\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [13-14]}).
	\end{proof}
	\begin{theorem}
		Soit $K$ un corps et soit $P(X) \in K[X]$, un polynôme unitaire et irréductible, alors Il existe une extension simple de  $K(\alpha)/K$, telle que $\alpha$ est algébrique sur $K$ de polynôme minimal $ irr_K(\alpha,X)=P(X)$.
	\end{theorem}
	\begin{proof}
		$K$ étant un corps donc l'anneau $K[X]$ est factoriel, donc $p(X)$ unitaire et irréductible est un élément premier non nul de $K[X]$, de plus $p(X)$ est maximal dans $K[X]$, ( $K[X]$ est un domaine principal ), donc on a ${\displaystyle L = \dfrac{K[X]}{(p(X))} }$ \, corps .
		
		Soit $\lambda$ l'injection canonique de $K$ dans $K[X]$ et soit $\delta$ la surjection canonique de $K[X]$ sur $L$, alors $\delta \circ \lambda$, un morphisme d'anneaux unitaire de $K$ dans $L$, est un monomorphisme.  On en déduit que $L$ est une extension de $K$, telle que $\forall \beta \in K$ est identifiable a son image par le monomorphisme $\delta \circ \lambda$.\\

		Dans $L$ posons $f(X) := \alpha$.
		\begin{equation*}
			L = \left\lbrace f(g(X)) \, ; \, g(X) \in K[X]\right\rbrace  = \left\lbrace g(\alpha) \, ; \, g(X) \in K[X] \right\rbrace .
		\end{equation*}
		D'autre part
		\begin{equation*}
			f(p(X)) = 0 \Longrightarrow p(\alpha) = 0 .
		\end{equation*}
		l'élément $\alpha$ de $L$ est nécessairement algébrique sur $K$ ce qui implique que $L = K(\alpha)$, on conclut que $L$ est une extension simple et algébrique de $K$, de plus par hypothèse $p(X)$ est unitaire et irréductible dans $K[X]$, donc
		\begin{equation*}
			p(\alpha) = 0 \Longrightarrow p(X) = irr_K(\alpha,X).
		\end{equation*}
	\end{proof}
	\begin{corollary}
		Tout extension simple et algébrique d'un corps $K$ est isomorphe a un corps de la forme $\dfrac{K[X]}{(p(X))}$, où $p(X)$ est un polynôme unitaire et irréductible dans $K[X]$.
	\end{corollary}
	\begin{example}\label{exmple1}
		Soit $p(X) = X^3 - 7$ dans $ \mathbb{Q} [X]$, le polynôme $p(X)$ est unitaire, irréductible car il est d'Eisenstein pour le nombre premier $t = 7$.\\
		Posons $\alpha = \sqrt[3]{7}$, on a bien $p(\alpha) = 0$ donc
		\begin{equation*}
			irr_K(\sqrt[3]{7},X) = X^3 - 7 \, ; \, \mathbb{Q}(\sqrt[3]{7}) \simeq \dfrac{\mathbb{Q}[X]}{(X^3 - 7)} \, ; \, [\mathbb{Q}(\sqrt[3]{7}) / \mathbb{Q}] = 3.
		\end{equation*}
		la famille $\left\lbrace 1 , \alpha , \alpha^2\right\rbrace $ forme une base de $\mathbb{Q}(\sqrt[3]{7})$ comme $\mathbb{Q}$-espace vectoriel et tout élément $x \in \mathbb{Q}(\sqrt[3]{7}) $ s'écrit d'une manière unique tel que on a
		\begin{equation*}
			x = a + b\sqrt[3]{7} + c(\sqrt[3]{7})^2 \, ; \, a,b,c \in \mathbb{Q}.
		\end{equation*}

	\end{example}
	\begin{definition}
		Soit $\alpha$ et $\beta$ deux élément de $K$ et algébriques sur $K$, Soit $K(\alpha)/K$, $K(\beta)/K$, deux extensions simples et algébriques de $K$. On dit que les éléments $\alpha$ et $\beta$ sont conjugué sur $K$, si
		\begin{equation*}
			irr_K(\beta,X) = irr_K(\alpha , X).
		\end{equation*}
		Dans ce cas, on dit aussi que les extensions $K(\alpha) / K$, $K(\beta) / K$ sont $K$-isomorphes.
	\end{definition}
	\begin{definition}
		On appelle corps de rupture d'un polynôme unitaire et irréductible  $p(X)$ sur $K$, le corps
		$K(\alpha)$, tel que $ p(X)=irr_K(\alpha,X) $.
	\end{definition}
	%
	
		\subsection{Extension algébrique}
	\begin{definition}
		On dit que l'extension $L /K$ est algébrique sur $K$ , si tout élément de $L$ est algébrique sur $K$
	\end{definition}
	\begin{theorem}
		Tout extension $L/K$ de degré fini, est nécessairement algébrique sur $K$.
	\end{theorem}
	\begin{proof}
		On pose $[L /K] = n \in \mathbb{N}^*$, et soit $\alpha \in L$.
		Dans le cas ou $\alpha \in K$, alors $\alpha$ est forcément algébrique sur $K$.
		Supposons maintenant que $n > 1$ et que $\alpha \in L\setminus K$, l'ensemble $\left\lbrace \alpha^i \;; i \in \mathbb{N} \right\rbrace $
		forme une base de $L$ comme $K$-espace vectoriel, par suite il existe un entier $t > 1$ et des élément $\beta_i, \; 0 \leq i \leq t$, non tous nul dans $L$, tels que
		\begin{equation*}
			\sum_{ 0 \leq i \leq t} \beta_i\alpha^i = 0 .
		\end{equation*}
		ce qui exprime que l'élément $\alpha$ est algébrique sur $K$.
	\end{proof}
	\begin{corollary}
		Tout extension simple, algébrique d'un corps $K$ est une extension algébrique de $K$.
	\end{corollary}
	\begin{theorem}
		L'extension $L /K$ est algébrique et de degré fini si et seulement si $L$ est obtenu par l'adjonction à  $K$ d'un nombre fini d'élément algébrique sur $K$.
	\end{theorem}
	
		\begin{proof}
	(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [18-19]}).
	\end{proof}

	\begin{theorem}
		Soient $L/K$ et $M/L$ deux extensions de corps; alors
		\begin{equation*}
			L /K \; et \; M/L \; \text{algébriques} \;\Longrightarrow M/K \; \text{algébrique}  .
		\end{equation*}
	\end{theorem}
\begin{proof}
(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [20]}).
	\end{proof}
\subsection{Clôtures algébriques}
\begin{proposition}
	Soit $K$ un corps ; les propriétés suivantes sont équivalents:
	\begin{enumerate}
		\item $K$ est algébriquement clos.
		\item Tout polynômes de $K[X]$ est scindé sur $K$.
		\item Tout polynômes irréductibles de $K[X]$ est de degré $1$.
	\end{enumerate}
\end{proposition}
\begin{proof}
(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [67] })
	
\end{proof}
\begin{example}
	\begin{itemize}
		\item Le corps $\mathbb{C}$ des nombres complexes est algébriquement clos.
		\item Le corps $\R$ des réels n'est pas algébriquement clos.
	\end{itemize}
\end{example}
\begin{theorem}
	Pour tout corps $K$, il existe au moins un corps $E$ algébriquement clos qui est extension de $K$.
\end{theorem}
\begin{proof}
	(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [72]})\end{proof}
\begin{example}
	\begin{itemize}
		\item Pour les deux corps $\mathbb{Q} \text{ et }\R$ il existe le corps $\Omega=\C$ algébriquement clos, et extensions de ces deux corps.
		\item Pour les corps finis de caractéristique $p$ (i.e., $\F_p$) il existe le corps $\Omega=\cup_{n\ge 0}\F_{p^{n!}}$ extension de $\F_p$ et algébriquement clos.
	\end{itemize}
\end{example}
\begin{definition}
	On appelle clôture algébrique $L$ de $K$ toute extension de $K$ telle que : 
	\begin{enumerate}
		\item $L$ est algébrique sur $K$.
		\item $L$ est algébriquement clos.
	\end{enumerate}
\end{definition}
\begin{theorem}
	Tout corps $K$ admet une clôture algébrique qu'on note $\overline{K}$. Plus précisément, si $L$ un corps algébriquement clos, extension de $K$ alors $$\overline{K}=\left\lbrace \al\in L\text{, }\al\text{ algébrique sur }K\right\rbrace\subseteq L $$ est un corps et c'est une clôture algébrique de $K$.
\end{theorem}
\begin{proof}
	(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [74] })
	
\end{proof}
\begin{example}
	\begin{enumerate}
		\item Nous savons déjà que $\C$ est un corps algébriquement clos (le théorème fondamentale de l'algèbre), et on sait que $\C$ est une extension algébrique sur $\R$ (car $\C=\R(i)$) alors $\C$ est la clôture algébrique de $\R$ et on écrit $\overline{\R}=\C$.
		\item Tandis que $\C$ n'est pas extension algébrique sur $\mathbb{Q}$ (car il existe en particulier $e$ et $\pi$ deux éléments transcendants sur $\mathbb{Q}$) donc $\C$ n'est pas une clôture algébrique de $\mathbb{Q}$. Mais comme $\C$ est algébriquement clos et contient $\mathbb{Q}$ alors $$\overline{\mathbb{Q}}=\left\lbrace \al\in\C\text{, }\al\text{ algébrique sur }\mathbb{Q}\right\rbrace \subsetneqq\C$$ est une clôture algébrique de $\mathbb{Q}$.
	\end{enumerate}
\end{example}
	
	\section{Extensions normales - Clôture normale}
	\subsection{Corps de décomposition d'un polynôme}
	Rappelons quelques définitions et résultats sur les polynômes à coefficients dans un corps $K$.
	\begin{itemize}
		\item[]{$\bullet$} Tout polynôme $f(X)$ de $K[X]$, de $deg\; n > 0$, admet exacteent $n$ racines, (distinctes ou confondus), dans $K$.
		\item[]{$\bullet$} dans l'anneau $K[X]$, tout polynôme $f(X) \in K[X]/K$ s'écrit d'une manière unique
		\begin{equation}\label{relationPoly}
			f(X) = a(p_1(X))^{n_1}(p_2(X))^{n_2}.\ldots.(p_r(X))^{n_r},\; \text{et} \; \sum_{1 \leq i \leq r} n_r = deg \;f 
		\end{equation}
		où $a \in K^*$, $r \in \mathbb{N}^*$, pour $1 \leq i \leq r$ les $p_i(X)$ sont des polynômes unitaires et irréductibles deux à  deux distincts dans $K[X]$.		
		\item[]{$\bullet$} Un polynôme $f(X)$ de $K[X]$ est dit scindé sur $K$, si dans la relation \eqref{relationPoly}, si $\forall i \; 1 \leq i \leq r$, $p_i(X)$ est de degré 1;
		\begin{equation*}
			f(X) = a(X-\alpha_1)^{n_1}(X-\alpha_2)^{n_2}\ldots (X-\alpha_r)^{n_r}.
		\end{equation*}
		\item[]{$\bullet$} Soit le polynôme $f(X) \in K[X]$, on suppose que $\alpha$ une racine de $f(X)$, ($f(\alpha) = 0$). le polynôme $f$, n'a que des racines simples $\iff f \land f' = 1$ et $f' \ne 0$, $f'$ désigne le polynôme dérivé de $f$.
		
	\end{itemize}
	%
	\begin{definition}
		Soit $f(X)$ un polynôme non constant dans $K[X]$, on appelle corps de décomposition de $f$ sur $K$ noté $Dec_k(f(X))$, tout extension $L$ de $K$ vérifiant
		\begin{enumerate} 
		\item $K \subseteq L \; et \; f(X) \text{est scindé sur}\, L$.
			
			\item $K \subseteq E \subseteq L \; et \; f(X) \text{est scindé sur}\, E \Longrightarrow L = E$.
		
		\end{enumerate}
	\end{definition}\label{def}
	\begin{proposition}
		Soit $L$ un corps de décomposition d'un polynôme $f(X)$ non constant de $K[X]$, si $n = deg\, f > 0$ et si $\beta_1,\beta_2,\ldots,\beta_n$ sont les racines distinctes ou confondues de $f$ dans $L$, alors, $L = K(\beta_1,\beta_2,\ldots,\beta_n)$, donc $L$ est une extension algébrique de degré fini sur $K$.
	\end{proposition}
	\begin{proof}
		On a \begin{equation}\label{eqa5}
			K \subseteq L \; et \; \forall i(1 \leq i \leq n), \, \beta_i \in L \Longrightarrow K(\beta_1,\beta_2,\ldots,\beta_n) \subseteq L.
		\end{equation}
		D'autre part le polynôme $f$ étant scindé sur $L$, et dans $L[X]$, $f(X)$ s'exprime par
		\begin{equation*}
			f(X) = a(X-\beta_1)(X-\beta_2)\ldots (X-\beta_n), \: \text{où} \, a \in K^*.
		\end{equation*}
		Donc d'après cette égalité $f$ est scindé sur $ K(\beta_1,\beta_2,\ldots,\beta_n)$, de plus la relation \eqref{eqa5} et la def.\ref{def} impliquent que $L = K(\beta_1,\beta_2,\ldots,\beta_n) $, on en déduit $L/K$ est une extension algébrique de degré fini.
	\end{proof}
	\begin{remark}
		La construction d'un corps de décomposition d'un polynôme non constant n'est pas unique.
	\end{remark}
	\begin{corollary}
		Deux corps de décompositions sur $K$ d'un polynôme $f(X)$ non constant dans $K[X]$, sont $K$-isomorphes.\\
		Le polynôme $f(x)$ étant scindé sur $L$, qui est extension du corps $K$, alors $L$ contient nécessairement un corps de décomposition de $f$ sur $K$.
	\end{corollary}
	\begin{remark}
		Le corps de décomposition d'un polynôme $f(X)$ non constant dans $K[X]$, est à  $K$-isomorphisme prés, la plus petit extension de $K$ sur laquelle $f(X)$ est scindé.
	\end{remark}
	\subsection{Extension Normale}
	\begin{definition}
		L'extension $L/K$ est dite normale si elle vérifie :
		\begin{enumerate}
			\item $L$ est algébrique sur $K$.
			\item Tout polynôme irréductible de $K[X]$, qui a une racine dans $L$ est scindé sur $L$.
			\item $L$ est corps de décomposition sur $K$, d'un polynôme de $K[X]$.
		\end{enumerate}
	\end{definition}
	
	\subsection{Clôture normale}
	\begin{definition}
		On appelle clôture normale d'une extension $L/K$, une extension $L'$ de $L$ telle que $K \subseteq L \subseteq L'$ telle que
		\begin{enumerate}
			\item $L'$ est extension normale de $K$.
			\item $(L \subseteq N \subseteq L' \; \text{extensions normales de} \, K) \Longrightarrow N = L'.$
		\end{enumerate}
	\end{definition}
	\begin{theorem}
		Pour tout extension de degré fini $L/K$ il existe une clôture normale $N$, de degré fini sur $K$ à  $K$-isomorphisme prés.
	\end{theorem}
	\begin{proof}
		(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [41] }).
	\end{proof}
	\begin{remark}
		Si $L = K(\alpha)$, $N$ est une clôture normale de $K(\alpha)$ sur $K$ si, et seulement si, $N$ est corps de décomposition sur $K$ du polynôme $irr_K(\alpha,X)$.
	\end{remark}
	\begin{example}\label{exemple2}
		On reprend l'exemple \ref{exmple1}, l'extension $\mathbb{Q}(\sqrt[3]{7})$, $\sqrt[3]{7} \in \mathbb{R}$, n'est pas une extension normale de $\mathbb{Q}$, car le polynôme $X^3 - 7$, irréductible sur $\mathbb{Q}$ n'a qu'une seule racine dans $\mathbb{Q}(\sqrt[3]{7})$.
		Cependant, si $j$ et $j^2$ désignent les racines cubiques non réelles de l'unité, dans $\mathbb{C}$, donc $N = \mathbb{Q}(\sqrt[3]{7},j\sqrt[3]{7},j^2 \sqrt[3]{7})$ est corps de décomposition de $irr_\mathbb{Q}(\sqrt[3]{7},X)$, donc $N$ est une clôture
		normale de $\mathbb{Q}(\sqrt[3]{7})$ sur $\mathbb{Q}$, contenue dans $\mathbb{C}$.
	\end{example}
	\section{Extensions séparables}
	\subsection{polynômes irréductibles, polynômes séparables}
	\begin{definition}
		On dit qu'un polynôme irréductible de $K[X]$ est \underline{séparable}, si il n'a que des racines simples dans un corps de décomposition sur $K$.
	\end{definition}
	\begin{proposition}
		Soit $f(X) \in K[X]\textbackslash K,$ et $f'(X)$ sont polynôme dérivé; alors
		\begin{enumerate}
			\item $car\, K = 0 \Longrightarrow f'(X) \ne 0$.
			\item $car\, K = p \ne 0 \Longrightarrow (f'(X) = 0 \iff f(X) = g(X^p))$, $g(X) \in K[X]$.
		\end{enumerate}
	\end{proposition}
	\begin{proof} On pose $d = deg \, f > 0$, donc dans $K[X]$
		\begin{equation*}
			(f(X) = \sum_{0 \leq i \leq d} \alpha_{i}X^i, \, \alpha_{i} \ne 0 ) \Longrightarrow f'(X) = \sum_{1 \leq i \leq d} i\alpha_{i}X^{i-1}.
		\end{equation*}
		\begin{enumerate}
			\item $car \, K = 0 \Longrightarrow \alpha_{i} \ne 0 \Longrightarrow d\alpha_{d} \ne 0 \Longrightarrow f'(X) \ne 0$ .
			\item $car \, K = p$, où $p$ est premier, l'expression de $f'(X)$ implique que
			\begin{equation*}
				\begin{array}{r c l}
					f'(X) = 0 & \iff & \forall i(1 \leq i \leq d), \, i\alpha_{i} = 0 \\
					& \iff &  \forall i(1 \leq i \leq d), \, (\alpha_{i} = 0 \, \text{ou} \, p | i ).
				\end{array}
			\end{equation*}
			On en déduit que, pour $r \in \mathbb{N^*}$, $\beta_k \in K, \, \forall 0 \leq k \leq r$.
			\begin{equation*}
				\begin{array}{r c l}
					f'(X) = 0 & \iff & f(X) = \sum_{0 \leq k \leq r } \beta_k X^{kp} \\
					& \iff &  f(X) = g(X^p), \: \text{où} \, g(X) = \sum_{0 \leq k \leq r } \beta_k X^k.
				\end{array}
			\end{equation*}
		\end{enumerate}
	\end{proof}
%
	\begin{definition}
		\begin{enumerate}
			\item Un polynôme $f(X) \in K[X] \textbackslash K$ est dit séparable sur $K$ si ses tout diviseurs irréductibles sont séparables sur $K$.
			\item Soit l'extension $L/K$, un élément $\alpha$ de $L$ est séparable sur $K$ si $\alpha$ est algébrique sur $K$ et le polynôme
			$irr_k(\alpha,X)$ est séparable sur $K$.
			\item L'extension $L/K$, est séparable si tout $\alpha$ de $L$ est séparable sur $K$.
			\item Un corps $K$ est dit \underline{parfait} si toute extension algébrique de $K$ est séparable sur $K$.
		\end{enumerate}
	\end{definition}
	\begin{remark}
		On remarque que tout extension séparable d'un corps $K$, est nécessairement algébrique sur $K$.
	\end{remark}
	\begin{theorem}
		\begin{equation*}
			K \, \text{corps parfait} \iff (car \,K = 0) \, ou \, (car \, K = p \, et \, K = \left\lbrace a^p ; \, a \in K \right\rbrace ).
		\end{equation*}
	\end{theorem}
	\begin{proof}
		(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [44-45] }).
	\end{proof}
	\begin{corollary}
		Tout corps fini de $car \, p \ne 0$ est parfait.
	\end{corollary}
	%
	\begin{corollary}
		Tout extension de $\mathbb{Q}$ est séparable.
	\end{corollary}
	%
	\begin{proposition}
		Soit $L/K$ une extension de degré fini, normale et séparable, alors $L$ est corps de décomposition sur $K$ d'un polynôme séparable.
	\end{proposition}
	\begin{proof}
		On suppose que $[L /K] > 1$, $L$ est normale et de degré fini sur $K$, donc $L$ est corps de décomposition d'un polynôme non constant $f(X)$ de $K[X]$, de plus $L$ est séparable sur $K$, ainsi tout facteur irréductible de $f(X)$ est scindé $L$, est séparable sur $K$, on en déduit que $f(X)$ est aussi séparable sur $K$.
	\end{proof}
	
	\begin{theorem}({\bf transitivité de la séparabilité})\\
		Soient $L/K$ et $M/L$ deux extension de corps,alors on a l'équivalence
		\begin{equation*}
			M/K \: \text{séparable} \Longrightarrow M/L \: et \: L/K \: \text{sont séparables}.
		\end{equation*}
	\end{theorem}
	\begin{proof}
		(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [45]}).
	\end{proof}
%
	\begin{theorem}{(Théorème de l'élément primitif)}\\
		Une extension finie $L/K$ est séparable si, et seulement si, elle est simple.
	\end{theorem}
	\begin{proof}
		(\textbf{Voir Calais J. Extensions de corps Théorie de Galois. Page [45-46]})
	\end{proof}
	%
	\begin{definition}
		Pour une extension $L/K$ séparable et de degré fini, on dit qu'un élément $\beta \in L$ est \underline{élément primitif}, si $L = K(\beta)$.
	\end{definition}
	
	\begin{definition}
		Étant donné une extension $L/K$, on dit qu'un automorphisme $\sigma$ du corps $L$ est un $K$-automorphisme de $L$, si $\sigma_{|K} = id_K$.
	\end{definition}
	%
	\subsection{Extension galoisienne}
	\begin{definition}\label{def2}
		On dit que l'extension $L/K$ est galoisienne si elle est a la fois normale et séparable.
	\end{definition}
	
	\begin{example}
		Dans l'exemple \ref{exemple2}, on a vu que l'extension, $\mathbb{Q}(\sqrt[3]{7},j) / \mathbb{Q}$, est normale, de plus $[\mathbb{Q}(\sqrt[3]{7},j) / \mathbb{Q}] = 3$, donc de degré fini, par suite la $car \, \mathbb{Q} = 0$, donc l'extension $\mathbb{Q}(\sqrt[3]{7},j) / \mathbb{Q}$ est séparable, on en déduit que c'est aussi une extension galoisienne d'après la définition \ref{def2}.
	\end{example}
		

	\chapter{Théorie de Galois}
\lhead{CHAPITRE 2}
\rhead{Théorie de  Galois}
La théorie de Galois est l'étude des extensions de corps $L\,/\,K$ au moyens des groupes des $K$-automorphismes de $L$.\\Cette méthode introduite par le mathématiciens français \textbf{Evarist Galois (1811-1832)} s'avéra d'une grande efficacité; elle lui permit, en particulier, de résoudre un problème qui préoccupait les mathématiciens de son époque, à savoir la caractérisation des équations polynomiales dites \textit{résolubles par radicaux}.\\Par ailleurs grâce à  la théorie de Galois, il a été possible d'apporter des réponses à  de nombreuses questions que se posaient les mathématiciens, parfois depuis l'Antiquité, telle la construction des polygones réguliers par la règle et le compas.

\section{Groupe de Galois et correspondance de Galois}
\begin{definition}
		On appelle \underline{groupe de Galois} d'une extension $L/K$, le groupe des $K$-automorphismes de $L$ qui sera noté $Gal(L / K)$.
	\end{definition}
	%
	\begin{theorem}
	\begin{itemize}
		\item Soit $\overline{K}$ une clôture algébrique de corps $K$ est soit $P$ un polynôme irréductible de $K[X]$. Pour toute racine $\alpha$  de $P$ dans $\overline{K}$, alors le nombre de $K$-automorphismes de $K(\alpha)$ est égale au nombre de racines distinctes de $P$ dans $\overline{K}$(i.e., de $K$-conjugués de $\alpha$ distincts) qui appartiennent à  $K(\alpha)$.
	\end{itemize}
\end{theorem}
\begin{proof}
	Considérons le schéma suivant
	\begin{equation}
		\xymatrix{
			\overline{K} \ar@{-}[d] & \overline{K} \ar@{-}[d]\\
			K(\alpha) \ar@{^{(}->>}[r]^{\sigma} \ar@{-}[d] & \sigma(K(\alpha)) \ar@{-}[d]\\
			K \ar@{^{(}->>}[r]^{id} & K
		}
	\end{equation}
	On applique le théorème de prolongement dans le cas particulier du schéma ci-dessus ; on a donc $P'=P$ et le nombre d'isomorphismes de $K(\alpha)$ dans $\overline{K}$.\\Il suffit de voir quels sont ceux qui donnent les automorphismes de $K(\alpha)$ ; or, si $\sigma$ est un tel $K$-isomorphisme, on rappelle que $\sigma(K(\alpha))=K(\sigma(\alpha))$, où $\sigma(\alpha)$ est une racine de $P$ dans $\overline{K}$, et on a donc $K(\alpha)=\sigma(K(\alpha))$ si et seulement si $\sigma(\alpha)\in K(\alpha)$ (en effet, si $\sigma(\alpha)\in K(\alpha)$, on a l'inclusion $K(\sigma(\alpha))\subseteq K(\alpha)$ qui conduit à  l'égalité car $[K(\alpha)/K]=[K(\sigma(\alpha))/K]=d(P)$, d'où $[K(\alpha)/K(\sigma(\alpha))]=1$).
\end{proof}
%
\begin{definition}\label{Corps fixe}
Soit $L/K$  une extension de groupe de Galois $G$. Pour tout sous-groupe $H$ de $G$ on définit le sous-corps des invariant de $L$, ou sous-corps fixé, par $H$, noté par  $L^H$ ou bien $Inv_L(H)$,  donné par : $Inv_L(H)=\left\{x\in L / \sigma(x)=x,\,\forall  \sigma \in H   \right\}$
\end{definition}
%
\begin{lemma}\label{lima1}
	Soit $L/ K$ une extension de corps de groupe de Galois $Gal(L / K)$
	\begin{enumerate}
		\item Si $K_1 \subseteq K_2$, sont des sous-corps de $L$ $\Longrightarrow Inv_L(Gal(L / K_2)) \subseteq Inv_L(Gal(L / K_2))$.
		\item Si $H_1 \subseteq H_2$ sont de sous-groupes de $Gal(L / K)$ $\Longrightarrow$ $Inv_L(H_2) \subseteq Inv_L(H_1)$.
		\item Si $H \subseteq Gal(L / K)$ $\Longrightarrow H \subseteq Gal(L  / Inv_L(H))$.
		\item Pour $H \leq Gal(L / K)$, si $K = Inv_L(H) \Longrightarrow K = Inv_L(Gal(L / K)). $
		\item Si $H = Gal(L / E)$, pour un sous-corps $K$ de $L$ $\Longrightarrow H = Gal(L  / Inv_L(H))$.
	\end{enumerate}
\end{lemma}
\begin{proof}
{\bf Voir [Lemme 2.9, Field and Galois Theory, Patrik Morandi, Graduate Texts in Mathematics book series (GTM, volume 167)] page[29-30]}
\end{proof}
\begin{example}
	\begin{itemize}
		\item Soit $K=\mathbb{Q}$ et $\overline{K}=\overline{\mathbb{Q}}\subset \mathbb{C}$, et soit $P(X)=X^2+1$. $P$ irréductible dans $\mathbb{Q}[X]$, car $P(X+1)=X^2+2X+2$ est d'Eisenstein en $p=2$. Soit $i$ une racine de $P$ dans $\mathbb{C}$ ; alors l'autre racine est $-i$ qui est dans $\mathbb{Q}(i)$. Le groupe des $\mathbb{Q}$-automorphismes de $\mathbb{Q}(i)$ est d'ordre $2$ (l'automorphisme non triviale étant ici est la conjugaison complexe).
		\item Toujours dans le même cadre, considérons $P(X)=X^3-2$ et soit $\sqrt[3]{2}$ la racine réelle de $P$ dans $\mathbb{C}$ ; comme $P$ est irréductible sur $\mathbb{Q}$ (car d'Eisenstein en $p=2$), les $\mathbb{Q}$-conjugués de $\sqrt[3]{2}$ sont $\sqrt[3]{2}$,$j\sqrt[3]{2}$,$j^2\sqrt[3]{2}$, où $j=\frac{-i+\sqrt{-3}}{2}$ est une racine primitive cubique de l'unité ; or $j\sqrt[3]{2}\notin \mathbb{Q}(\sqrt[3]{2})$, sinon on aurait $j\in \mathbb{Q}(\sqrt[3]{2})$, ce qui est absurde car $\mathbb{Q}(\sqrt[3]{2})\subset \mathbb(R)$ tandis que $j\notin \mathbb(R)$. Dans ce cas, le groupe des $\mathbb{Q}$-automorphismes est réduit Ã  l'identité, bien que $P$ ait trois racines distinctes.\\Noter cependant qu'il existe trois $\mathbb{Q}$-isomorphismes distincts de $\mathbb{Q}(\sqrt[3]{2})$ dans $\mathbb{C}$, ce qui fournit les trois corps distincts : $\mathbb{Q}(\sqrt[3]{2})$, $\mathbb{Q}(j\sqrt[3]{2})$ et $\mathbb{Q}(j^2\sqrt[3]{2})$.
	\end{itemize}
\end{example}
\begin{lemma}\label{Lem corps fixes}
	\begin{itemize}
		\item (i) Soit $L/K$ une extension galoisienne de groupe de Galois $G$, et soit $K'=L^G$; alors $K'=K$.
		\item (ii) si $F/K$ est une sous-extension de $L/K$, alors $L/F$ est une extension galoisienne et:
		\begin{equation}
			Gal(L/F)=\left\lbrace \sigma \in G,\,\sigma(x)=x\,pour\,tout\,x\in F\right\rbrace
		\end{equation}
	\end{itemize}
\end{lemma}
\begin{proof}
	(i) On a $K\subseteq K$ puisque les éléments de $G$ sont des $K$-automorphismes. Soit alors $\sigma\in G$ ; on constate que la restriction de $\sigma$ à $K'$ est l'identité (utiliser la définition de $K'$), donc tout élément de $G$ est un $K'$-automorphisme de $L$, d'où $Gal(L/K)\subseteq Gal(L/K')$, ce qui conduit Ã  $[L/K]\leq[L/K']$ ; d'où $K'=K$ d'après un raisonnement classique sur le degré lorsque $K\subseteq K'\subseteq L$
\end{proof}

	\end{enumerate}

Dans l'exemple précédent le diagramme des sous-corps de $L$ peut s'établir alors sans difficulté ; cependant on voit que dans un cas plus compliqué il est utile de préciser les inclusions ; or ceci résulte en fait de \ref{Th1CG} qui contient le résultat plus précis suivant.
\begin{corollary} \label{corolaire CG}
	Soit $L/K$ une extension galoisienne finie de groupe de Galois $G$ ; la correspondance de Galois
	$$
	F \mapsto Gal(L/F) \hspace*{5mm} (K\subseteq F\subseteq L)\text{,}
	$$
	de l'ensemble des sous-extension $L/K$ dans l'ensemble des sous-groupes de $Gal(L/K)$, a les propriétés de décroissance suivantes (pour l'inclusion)
	\begin{enumerate}
		\item $F_1 \subseteq F_2 \iff Gal(L/F_1) \supseteq Gal(L/F_2)$
		\item $Gal(L/F_1F_2)=Gal(L/F_1) \cap Gal(L/F_2)$
		\item $Gal(L/F_1 \cap F_2)=\left\langle Gal(L/F_1),Gal(L/F_2)\right\rangle $
	\end{enumerate}
\end{corollary}
\begin{proof}
	On a
	\begin{equation*}
		F_1\subseteq F_2 \Longrightarrow Gal(L/F_1) \subseteq Gal(L/F_2)
	\end{equation*}
	(par définition de $F_i$-automorphisme, $i=1,2$), puis:
	\begin{equation*}
		Gal(L/F_1) \subseteq Gal(L/F_2) \Longrightarrow L^{Gal(L/F_1)} \subseteq L^{Gal(L/F_2)}
	\end{equation*}
	(Par la définition \ref{Corps fixe}), d'où $F_1\subseteq F_2$. D'où le point $1.$
	Le point $2.$ est également immédiat en vérifiant chaque inclusion.
	Pour le point $3.$, soit $H$ un sous-groupe de $G$ contenant $H_1=Gal(L/F_1)$ et $H_2=Gal(L/F_2)$, et soit $F$ le corps fixe par $H$. On a donc (point précédents) $F\subseteq F_1$ et $F\subseteq F_2$ ; ainsi $F\subseteq F_1 \cap F_2$, d'où $Gal(L/F_1\cap F_2)\subseteq H$. En passant Ã  l'intersection sur les $H$ (contenant $H_1$ et $H_2$), on obtient :
	\begin{equation}
		\large{Gal(L/F_1\cap F_2)\subseteq \bigcap_{H\supseteq H_1, H_2} H=\left\langle H_1,H_2\right\rangle }
	\end{equation}
	Réciproquement si $\sigma\in \left\langle H_1,H_2\right\rangle$, il est claire, en utilisant le théorème sur les sous-groupes engendrés par une partie, que $\sigma$ laisse fixe les éléments de $F_1\cap F_2$ ; on a donc $\left\langle H_1,H_2\right\rangle \subseteq Gal(L/F_1\cap F_2)$, d'où l'égalité.
\end{proof}
Enfin, une fois le treillis des sous-extensions $F/K$ de $L/K$ établi, on peut se demander quelles extensions $F/K$ sont galoisiennes ; on a Ã  ce sujet le deuxième énoncés fondamental suivant :

			\sigma & \mapsto & \sigma_{|F}

	\end{equation}
	Dont le noyau est $H$.
\end{theorem}

\begin{theorem} \label{Th3CG} 
	Soit $L/K$ et $F/K$ deux sous-extensions, finies, galoisiennes, de $\overline{K}/K$. Posons successivement $G=Gal(LF/K)$, $H_F=Gal(LF/F)$ et $H_L=Gal(LF/L)$. Alors $F\cap L/K$ et galoisienne et on a
	\begin{equation}
		Gal(L\cap F/ K)\simeq G/H_LH_F
	\end{equation}
\end{theorem}
\begin{proof}
	Puisque $L/K$ et $F/K$ sont galoisiennes, $H_L$ et $H_F$ sont des sous-groupes normaux de $G$. $\left\langle H_L,H_F\right\rangle = H_LH_F$ est normal dans $G$ ; d'où le résultat par le 3. du corollaire \ref{corolaire CG}.
\end{proof}
\begin{theorem} \label{Th4CG}
	Soit $K$ un corps, et soient $L$ et $L'$ deux sous-extensions d'une extension non nécessairement algébrique de $K$. On suppose que $L/ L\cap L'$ est finie galoisienne.\\ Alors l'extension $LL'/ L'$ est finie galoisienne, et l'application :
	\begin{equation}
		\xymatrix{
			Gal(LL'/L') & \longrightarrow & Gal(L/ L\cap L')\\
			\sigma &\mapsto &\sigma_{|L}\\
		}
	\end{equation}
	est isomorphisme canonique de groupes.
\end{theorem}
\begin{proof}
	L'application écrite a un sens, car si $\sigma$ est un $L'$-automorphisme de $LL'$, par restriction à  $L$, il laisse fixe $L\,\cap\, L'$. Si $\sigma_{|L}$ est l'identité, c'est que $\sigma$ est l'identité sur $L$ et $L'$, donc sur $LL'$, d'où l'injectivité, $L$ et $L'$ sont linéairement disjointes sur $L\,\cap \,L'$, d'où : $$\left[ LL'/L'\right] = \left[ LL/L\,\cap\,L'\right]$$ ainsi le fait que l'application étudiée soit injective suffit Ã  prouver qu'elle est surjective.\\
	Ce qui précède donne lieu au schéma suivant :
	\begin{equation}
		\xymatrix{
			&LL'\ar @{-}[d]\\
			L\ar @{-}[ru] \ar @{-}[d] & L'\\
			L\,\cap \, L' \ar @{-} [ru] \ar @{-} [d]\\
			K
		}
	\end{equation}
	dans lequel on a $Gal(LL'/L')\,\simeq\, Gal(L/L\,\cap \, L')$ par restriction.
\end{proof}
\begin{theorem}\label{Th5CG}
	Soient $F$ et $L$ des sous-extensions de $\overline{K}/K$, galoisienne sur $F\,\cap\,L$. L'application canonique :
	\begin{equation}
		\xymatrix{
			Gal(FL/F\,\cap \, L) & \longrightarrow & Gal(F/F\,\cap \, L) \times Gal(L/F\,\cap \, L)\\
			\sigma & \mapsto & (\sigma_{|F}\,,\,\sigma_{|L})
		}
	\end{equation}
	est un isomorphisme canonique de groupe.
\end{theorem}
\begin{proof}
	On sait que $FL/F\cap L$ est galoisienne ; comme les extensions $F/F\,\cap \, L$ et $L/F\,\cap \, L$ sont linéairement disjointes, on a $\left[ FL/F\cap L \right]=\left[ F/F\cap L \right]\left[ L/F\cap L \right]$ et il suffit de montrer l'injectivité.\\
	Le schéma est alors le suivant, les branches parallèles correspondant Ã  des groupes de Galois isomorphes (c'est le théorème \ref{Th4CG} appliqué deux fois), et où le groupe $Gal(FL/F\,\cap \, L)$ peut aussi s'écrire $Gal(FL/F) \times Gal(FL/L)$ :
	\begin{equation*}
		\xymatrix{
			&FL\ar @{-}[d]\\
			F\ar @{-}[ru] \ar @{-}[d] & L\\
			F\,\cap \, L \ar @{-} [ru] \ar @{-} [d]\\
			K
		}
	\end{equation*}
	\textbf{Attention} $F\,\cap \, L/K$ n'est pas nécessairement galoisienne car les hypothèses sont plus faibles qu'en théorème \ref{Th3CG}.
\end{proof}
\section{Groupe de Galois des polynômes}
\subsection{Norme et trace}
Soit $L/K$ une extension de degré fini $n = [L / K]$, pour tout $\lambda$ dans $L$ on définit l'application noté $M_\lambda$ définie par
\begin{equation*}
	\begin{array}{r c l}
		M_\lambda : L & \to & L \\
		x & \mapsto & \lambda x
	\end{array}
\end{equation*}
$M_\lambda$ définie ainsi un endomorphisme d'espace vectoriel $L$, on l'appelle endomorphisme multiplication par $\lambda$.
\begin{definition}
	On appelle trace, respectivement norme, d'un élément $\lambda$ de $L$, noté $Tr_{L/K}(\lambda)$, respectivement $N_{L/K}(\lambda)$, la trace et le déterminant de l'endomorphisme $M_\lambda$. On écrit alors
	\begin{equation*}
		Tr_{L/K}(\lambda) = Tr(M_\lambda) \; et \; N_{L/K}(\lambda) = Det(M_\lambda).
	\end{equation*}
\end{definition}
\begin{example}
	Soit l'extension $L/K$, où $K = \mathbb{Q}$ et $L = \mathbb{Q}(\sqrt[3]{2})$, on veut déterminer la trace et la norme de l'élément $\lambda = \sqrt[3]{2}$. on sait que la famille $\left\lbrace 1 , \sqrt[3]{2} , \sqrt[3]{2^2}\right\rbrace = \mathbf{F}$, forme une base de $L$ comme $K$-espace vectoriel, alors l'image de chacun des éléments de cette base par endomorphisme multiplication par $\lambda$ est
	\begin{equation*}
		M_\lambda(1) = \lambda 1 = \sqrt[3]{2} \;, M_\lambda(\sqrt[3]{2}) = \lambda \sqrt[3]{2} = \sqrt[3]{2^2} \;, M_\lambda(\sqrt[3]{2^2}) = \lambda \sqrt[3]{2^2} = 2
	\end{equation*}
	Ainsi la matrice associée a l'endomorphisme $M_\lambda$ relativement a la base $\mathbf{F}$ est
	\begin{equation*}
		\begin{pmatrix}
			0 & 0 & 2 \\ 1 & 0 & 0 \\ 0 & 1 & 0
		\end{pmatrix}
	\end{equation*}
	Ce qui donne $$Tr_{L/K}(\lambda) = 0 \; et \; N_{L/K}(\lambda) = 2$$
\end{example}
\begin{lemma}
	$L/K$ est une extension finie de degré $n$, alors
	\begin{itemize}
		\item Soit $x$ et $y$ dans $L$, alors $Tr_{L/K}(x + y) = Tr_{L/K}(x)+Tr_{L/K}(y)$ et $N_{L/K}(xy) = N_{L/K}(x)N_{L/K}(y)$.
		\item si $a \in K$, alors $Tr_{L/K}(a) = na$ et $N_{L/K}(a) = a^n$ sont dans $K$.
	\end{itemize}
\end{lemma}
\begin{proposition}
	$L/K$ est une extension finie de degré $n$, pour $\alpha$ dans $L$ notons
	\begin{equation*}
		p(X) = X^m + a_{m-1}X^{m-1} + \ldots + a_1X + a_0 \in K [X].
	\end{equation*}
	Son polynôme minimal, alors $N_{L/K}(\alpha) = (-1)^na_0^{nm}$ et $Tr_{L/K}(\al) = - \frac{n}{m}a_{m-1}$.
\end{proposition}
\begin{proof}
	Par hypothèse $[K(\al) / K] = m$, $K[\al] \subseteq L$ donc $m$ divise $n = [L / K]$. \\
	On considère l'endomorphisme $M_\al$, on note $g(X)$ le polynôme caractéristique de $M_\al$, soit
	$$g(X) = X^n + b_{n-1}X^{n-1} + \ldots + b_1X + b_0 \in K [X] $$  \\
	$g(X)$ et $p(X)$, on le même facteur irréductible donc $g(X)$ est une puissance du polynôme minimal de $\alpha$, ce qui donne $$g(X) = (p(X))^{nm}.$$
	Comme $N_{L/K}(\al) = Det(M_\al) = (-1)^nb_0$ et $Tr_{L/K}(\al) = -b_{n-1}$, on retient que $g(X) = (p(X))^{n/m}$, on aura $$b_0 = a_0^{nm} \; et \; b_{n-1} = \frac{n}{m}a_{m-1}$$.
\end{proof}
\begin{corollary}
	$L/K$ est une extension finie de degré $n$, pour $\alpha$ dans $L$ notons par $\sigma_i; \, 1 \leq i \leq r$, les $r$ $K$-homomorphisme de $K(\al)$ dans
	une clôture algébrique de $K$, alors
	\begin{equation*}
		N_{L/K}(\al) = \left( \prod_{i = 1}^{r} \sigma_i(\al)\right) ^{n/r} \: et \: Tr_{L/K}(\al) = \dfrac{n}{r} \left( \sum_{i = 1}^{r} \sigma_i(\al) \right) .
	\end{equation*}
\end{corollary}
\begin{proof}
	Soit $\Omega$ une clôture algébrique de $K$, alors le polynôme minimal de tout $\alpha$ dans $L$ est
	\begin{equation*}
		p(X) = \prod_{i=1}^{r} (X - \sigma_i(\al)) \in K[X].
	\end{equation*}
	où les $\sigma_i(\al)$ dans $\Omega$, on sait que le polynôme caractéristique $$g(X) = X^n + b_{n-1}X^{n-1} + \ldots + b_1X + b_0 \in K [X] $$
	de l'endomorphisme $M_\al$ est lié au polynôme minimal $p(X)$ par la relation $g(X) = (p(X))^{n/r}$. \\
	D'autre parte $$N_{L/K}(\al) = (-1)^nb_0 \; et \; Tr_{L/K}(\al) = -b_{n-1}.$$
	Avec $$b_0 = \prod_{i = 1}^{r} \sigma_i(\al) \; et \; b_{n-1} = \dfrac{n}{r} \left( \sum_{i = 1}^{r} \sigma_i(\al) \right).$$
\end{proof}
\begin{corollary}
	Soit $L/K$ une extension galoisienne de groupe de Galois $G = Gal(L / K)$, alors pour tout $\alpha \in L$ on a
	\begin{equation*}
		N_{L/K}(\al) = \prod_{\sigma \in G} \sigma(\al) \; et \; Tr_{L/K}(\al) = \sum_{\sigma \in G} \sigma(\al).
	\end{equation*}
\end{corollary}
\subsection{Discriminant}
\begin{definition}
	Soit $L/K$ une extension finie de corps séparable de degré $n$, on appelle discriminant d'une famille finie $\left( \al_i \right)_{1 \leq i \leq n},$ d'élément de $L$.\\
	L'élément de $L$ noté $D_{L/K}(\al_1,\ldots,\al_n) = (Det(\sigma_i(\al_j)))^{2} \,, 1 \leq j \leq n$, où les $\sigma_i$ sont les $n$ plongement de $L$ dans une clôture algébrique de $K$.
\end{definition}
\begin{remark}
	$$D_{L/K}(\al_1,\ldots,\al_n) = (Det(Tr_{L/K}\al_i\al_j) \; et \; D_{L/K}(\al_1,\ldots,\al_n) \in K.$$
\end{remark}
\begin{proposition}
	Soit $L/K$ une extension finie de corps séparable de degré $n$, si $L = K(\al),$ alors
	\begin{equation*}
		D_{L/K}(1,\al,\ldots,\al^{n-1}) = (-1)^{n(n-1)/2} \prod_{j \ne i} (\sigma_i(\al) - \sigma_j(\al)) = (-1)^{n(n-1)/2} N_{L/K}(p'(\al)).
	\end{equation*}
	où les $\sigma_i :\; 1 \leq i \leq n,$ sont les $n$ plongements de $L$ dans une clôture algébrique $\Omega$ et $p'$ le polynôme dérivé du polynôme minimal $p$ de $\alpha$.
\end{proposition}
\begin{proof}
	On a bien $$D_{L/K}(1,\al,\ldots,\al^{n-1}) = (Det(\sigma_i(\al^j)))^2.$$ D'autre part $\sigma_i(\al^j) = \sigma_i(\al)^j$ et que la matrice $\left( \sigma_i(\al)^j\right)$ est une matrice de "Vandermonde" de $n$-uplet $\sigma_1(\al),\ldots,\sigma_n(\al)$ dont le déterminant est $\prod_{i < j}(\sigma_i(\al) - \sigma_j(\al))$ de qui donne
	\begin{equation*}
		D_{L/K}(1,\al,\ldots,\al^{n-1}) = \left( \prod_{i < j}(\sigma_i(\al) - \sigma_j(\al))\right) ^2.
	\end{equation*}
	Chaque différence $\sigma_i(\al) - \sigma_j(\al)$ est associée a un couple $(\sigma_i(\al),\sigma_j(\al))$; donc on trouve\\
	$n(n-1)/2$ de ces différences
	dans le produit $\prod_{i < j}(\sigma_i(\al) - \sigma_j(\al))$ et donc $n(n-1)$ de ces différences dans $$\left( \prod_{i < j}(\sigma_i(\al) - \sigma_j(\al))\right) ^2$$, on trouve le même résultat dans $\prod_{i \ne j}(\sigma_i(\al) - \sigma_j(\al))$; ainsi, on a bien
	\begin{equation*}
		D_{L/K}(1,\al,\ldots,\al^{n-1}) = \left( \prod_{i \ne j}(\sigma_i(\al) - \sigma_j(\al))\right).
	\end{equation*}
	D'autre part, on a $p = \prod_{i = 1}{n}(X-\sigma_i(\al)),$ alors le polynôme dérivé $p'$ s'écrit
	\begin{equation*}
		p'(X) = \sum_{j=1}^{n}\prod_{i \ne j}(X-\sigma_i(\al)).
	\end{equation*}
	Ce qui entraine $\forall i, \, 1 \leq i \leq n$, on a
	\begin{equation*}
		\sigma_i(p'(\al)) = p'(\sigma_j(\al)) = \prod_{i \ne j}(\sigma_i(\al) - \sigma_j(\al)).
	\end{equation*}
	Donc $p'(\al)$ s'écrit comme suite
	\begin{equation*}
		N_{L/K}(p'(\al)) = \prod_{i = 1}^{n}\sigma_i(p'(\al)) = \prod_{i \ne j}(\sigma_i(\al)-\sigma_j(\al)).
	\end{equation*}
	On en déduit alors
	\begin{equation*}
		D_{L/K}(1,\al,\ldots,\al^{n-1}) = (-1)^{n(n-1)/2} N_{L/K}(p'(\al)).
	\end{equation*}
\end{proof}
\begin{definition}
	Soit $K$ un corps de caractéristique différente de $2$, $p(x) \in K[X]$ un polynôme de degré $n$, ayant $n$ racines distinctes $\left( \al_i\right)_{1\leq i \leq n}$ dans une clôture algébrique $\Omega$ de $K$. On appelle discriminant de $p$, l'élément noté $D_p$ de $\Omega$ défini par
	\begin{equation*}
		D_p = \left( \prod_{1 \leq i \leq j \leq n}(\al_i - \al_j)\right)^2
	\end{equation*}
	On note $D_p = \Delta^2$, où $\Delta = \left( \prod_{1 \leq i \leq j \leq n}(\al_i - \al_j)\right) $.
\end{definition}
\begin{remark}
	Le discriminant $D_p$ du polynôme $p$ est un élément de $K$.
\end{remark}
\begin{example}\label{exempledis}
	Soit $K$ un corps, avec $car \, K \ne \left\lbrace  2 , 3 \right\rbrace $, soit $\chi(X) = X^3 + pX + q \in K[X]$ un polynôme irréductible sur $K$, on note $\alpha$ une racine de $\chi(X)$ et soit $L = K(\al)$, donc le discriminant de $\chi$ est
	\begin{equation*}
		D_\chi = -N_{L/K}(\chi'(\al))
	\end{equation*}
	avec $\chi'(\al) = 3\al^2 + p$ et tel que $\alpha^3 + p\al + q = 0$. \\
	On sait que $N_{L/K}(\chi'(\al))$ est le déterminant de la matrice de l'endomorphisme $M_{\chi'(\al)}$, multiplication par $\chi'(\al)$, de $L$ dans la base est $\left\lbrace 1,\al,\al^2 \right\rbrace $ du $K$ espace vectoriel $L$ on a donc
	\begin{equation*}
		M_{\chi'(\al)}(1) = p + 3\al^2, \; M_{\chi'(\al)}(\al) = -3q - 2p\al, \; M_{\chi'(\al)}(\al^2) = 3\al - 2p\al^2.
	\end{equation*}
	D'où la matrice suivante
	\begin{equation*}
		\begin{pmatrix}
			p & -3q & 0 \\
			0 & -2p & -3q \\
			3 &  0  & -2p
		\end{pmatrix}
	\end{equation*}
	dont le déterminant est $4p^3 + 27q^2$, on en déduit que $D_\chi = -(4p^3 +27q^2)$
\end{example}

\subsubsection{Polynôme de degré 4}
\textbf{Transformation (méthode de Ferrari) :}
Soit $K$ un corps de caractéristique différent de $2$ et $3$, $a_1,a_2,a_3,a_4 \in K$, $g(X) = X^4 - a_1X^3 + a_2X^2 - a_3X + a_4$.\\
On pose $Y = X - \frac{s_1}{4}$, (transformation de Tschirnhaus), $g(X)$ devient $f(Y) = Y^4 + pY^2 + qY + r , \, p,q,r \in K$.\\
La méthode consiste a mettre $f$ sous la forme : $f(Y) = (Y^2 + uY + v)(Y^2 + u'Y + v')$, on suppose que $q \ne 0$ (car sinon on aura une équation bicarré).\\
Par suite l'identification nous donne le système :
\begin{equation*}
	\left\lbrace \begin{array}{r c l}
		u + u' & = & 0 \\
		uu'+ v+ v' &=& p \\
		uv' + u'v & = & q \\
		vv' & = & r
	\end{array} \right.
	\iff
	\left\lbrace \begin{array}{r c l}
		u' & = & u \\
		v+ v' &=& p+ u^2 \\
		v + v' & = & \dfrac{q}{u} \\
		vv' & = & r
	\end{array} \right.
\end{equation*}
Donc, on en déduit que
\begin{equation*}
	\left\lbrace 	\begin{array}{r c l}
		v & = & (\frac{1}{2u}) \left( u(p+u^2) - q \right) \\
		v' & = & ( \frac{1}{2u}) \left( u(p+u^2) + q \right).
	\end{array} \right.
\end{equation*}
Par suite $vv' = r$ donne $$u^2(u^2+p)^2 - q^2 = 4ru^2.$$
On pose $z = u^2 +p$, on en déduit que $z$ est racine de l'équation du troisième degré
\begin{equation}\label{E1}
	Z^3 - pZ^2 - 4rZ + 4rp - q^2.
\end{equation}
L'équation \eqref{E1} est dite cubique résolvante de l'équation
\begin{equation}\label{E2}
	Y^4 + pY^2 + qY + r = 0.
\end{equation}
la résolution de \eqref{E1} donne $z$ et donc $u$ et par suite $v$ et $v'$. \\
L'équation \eqref{E2} est donc ramenée Ã  la résolution du système suivant
\begin{equation*}
	\left\lbrace 	\begin{array}{r c l}
		(E_1) \; Y^2 + uY + v &= &0 \\
		(E_2) \; Y^2 + u'Y + v' &= & 0
	\end{array} \right.
	\longmapsto
	\left\lbrace 	\begin{array}{r c l}
		D_{(E_1)}  & = & u^2 - 4v \\
		D_{(E_2)} &= & u'^2 - 4v'.
	\end{array} \right.
\end{equation*}
\textbf{Formules connues : } Soit $p,q,r,t \in K$, on pose $p(X) = X^4 + pX + q$ et $f(X) = X^4 + rX^2 + t$, alors
\begin{equation*}
	\begin{array} {r c l}
		D_p &=& -27p^4 + 256q^3 \\
		D_f &=& 16t(r^2 - 4t)^2
	\end{array}
\end{equation*}
\subsubsection{Polynôme de degré n ($\mathbf{X^n + pX + q}$) : }
Soit $p(X) = X^n + aX +b , \, a,b \in K$, soit $\al := \al_1,\al_2,\ldots,\al_n$ ses $n$ racines distinctes dans une clôture algébrique de $K$, on note que $$p(X) = \prod_{i=1}^{n} (X - \al_i) $$ \\
Posons $p(\al_i) = \al_i^n + a\al_i + b$, on a $p(\al_i) = 0, \; \forall 1 \leq i \leq n$, donc $$\al_i^n + a\al_i + b = 0 \iff \al_i^n = -a\al_i - b$$
Donc si $p'$ désigne le polynôme dérivé de $p$ alors on a, $$p'(\al_i) = n\al_i^{n-1} + a = n(\frac{\al_i^n}{\al_i}) + a = \dfrac{n(-a\al_i -b) + a\al_i}{\al_i} = \dfrac{(1-n)a\al_i - nb}{\al_i}.$$
Soit \begin{equation}\label{equationt}
	N(p'(\al_i)) =\prod_{i = 1}^{n} p'(\al_i) = \prod_{i = 1}^{n} \dfrac{(1-n)a\al_i - nb}{\al_i} = \dfrac{\prod_{i = 1}^{n} (1-n)a\al_i - nb }{\prod_{i = 1}^{n} \al_i}  .
\end{equation}
Pour venir about de ce calcul on considère l'égalité suivante $$(1-n)a\al_i - nb = a(1-n)\al_i -nb = -a(1-n)(\frac{nb}{a(1-n)}-\al_i)$$on pose alors $q = \dfrac{nb}{a(1-n)}$ et on remplace dans l'équation \ref{equationt}, on a donc
\begin{equation*}
	\dfrac{\prod_{i = 1}^{n} (1-n)a\al_i - nb }{\prod_{i = 1}^{n} \al_i} = \dfrac{(-1)^n a^n (1-n)^n \prod_{i = 1}^{n} (q-\al_i) }{\prod_{i = 1}^{n} \al_i} =
	\dfrac{(-1)^n a^n (1-n)^n p(q) }{(-1)^nb} = \dfrac{a^n (1-n)^n p(q) }{b}.
\end{equation*}
On pose $t = \dfrac{a^n(1-n)^n}{b}$ donc on a bien $$ N(p'(\al_i)) = tf(q)$$.\\
Sachant que $D_p = (-1)^{\frac{n(n+1)}{2}}N(p'(\al_i))$ donc le discriminant de $p$ est donné par $$(-1)^{\frac{n(n+1)}{2}}\left( tf(q)\right) , \; avec \;t = \dfrac{a^n(1-n)^n}{b}, \; et \; q = \dfrac{nb}{a(1-n)} $$
\begin{example}
	Soit $p(X) = X^5 + 5X + 1 \in \mathbb{Q}[X]$, le discriminant de $p$ d'après le résultat précédent est donné par
	\begin{equation*}
		D_p = (-1)^{\frac{5(5+1)}{2}}(\frac{5^5(1-5)^5}{1})p(\dfrac{5}{5(1-5)}) = 5^5 + 5^6(-4)^4 -5^5(4)^5 = 800005.
	\end{equation*}
\end{example}


\subsection{Groupe de Galois d'un polynôme}
\begin{definition}
	Pour Tout polynôme $P(X) \in K[X]$, séparable sur $K$, et de corps de décomposition $L$ sur $K$, le groupe de Galois du polynôme $P$, noté
	$Gal(P/K)$ est le groupe de Galois $G = Gal(L / K)$ de l'extension galoisienne $L/K$.
\end{definition}
Notons que si $P(X) \in K[X]$ un polynôme irréductible sur $K$ de degré $n$ de groupe de Galois $G$, alors pour tout $\sigma \in G$, et pour tout racine $\alpha$ de $P(X)$, l'élément $\sigma(\al)$ est aussi racine de $P(X)$, En d'autre terme, si on a $\alpha$,$\beta$ deux racines de $P(X)$, $\exists \sigma \in G$ tel $\sigma(\al) = \beta.$ On dira que $\sigma$ agit sur les racines du polynôme $P(X)$ par permutation de celles-ci et cette action est dite transitive.
\begin{definition}
	On dit qu'un sous-groupe de $G$ de permutation du groupe $S_n$ est transitif si, $$\forall i \ne j ; \; 1 \leq i , j \leq n, \exists \sigma \in G ; \; \sigma(i) =j.$$
\end{definition}
\begin{theorem}\label{thoreme3}
	Soit $p(X) \in K[X]$ un polynôme de degré $n$ et de groupe de Galois $G$ alors
	\begin{enumerate}
		\item $G$ peut être identifié à sous-groupes du groupe symétrique $S_n$.
		\item Si $p(X)$ irréductible, alors $n \mid |G|$ et $G$ est isomorphe à un sous-groupe transitif de $S_n$.
	\end{enumerate}
\end{theorem}
\begin{proof}
	Notons par $\al := \al_1,\ldots,\al_n$ les différentes racines de $p(X)$ dans un corps de décomposition $L$ de celui-ci, $deg \, p \geq n \geq 1$.
	\begin{enumerate}
		\item Tout $\sigma \in G$ induit une permutation de l'ensemble $\left\lbrace \al_1,\ldots,\al_n \right\rbrace $ des racines de $p(X)$, ce qui définit un homomorphisme $\varPhi : G \to S_n$ qui est injectif, ainsi donc $G \simeq Im\varPhi$ sous-groupes de $S_n$.
		\item Si $p(X)$ est irréductible, alors pour toute raine $\alpha$ de $p(X)$, dans une clôture algébrique de $K$, $[K(\al) / K] = n = deg \, p.$ d'autre part $K(\al) \subseteq L$; où $L$ est corps de décomposition de $p(X)$, on en déduit que $n \mid |G| = [L / K]$.\\
		De plus $\forall i,j \; , 1 \leq i,j \leq n \: (i \ne j)$, il existe un $K$-isomorphisme $\sigma : K(\al_i) \to K(\al_j)$, donné par
		$\sigma(\al_i) = \al_j$, qui se prolonge en un $K$-automorphisme de $L$, ce qui montre que $G$ est transitif, $G \simeq S_n.$
	\end{enumerate}
\end{proof}
\begin{example}
	\label{exempleder} Soit $K$ un corps de caractéristique différente
	de $2$ et soit $f(X)\in K[X]$ un polynôme irréductible de degré
	$2$, on pose $K(\al)=L$ avec $\alpha$ une racine de $f(X)$ dans
	une clôture de $K$, soit alors le groupe de Galois $G=Gal(L/K)$,
	$f(X)$ est séparable car la $car\,K\ne2$, si les racines de $f(X)$
	sont dans $K$ alors il est évident que $L=K$, ce qui montre que
	$G=id_{K}$, sinon si les racines de $f(X)$ ne sont pas dans $K$,
	alors $|G|=2=|S_{2}|$ et donc $G\simeq S_{2}\simeq\mathbb{Z}/2\mathbb{Z}$. 
\end{example}

\section{Groupes de Galois des extensions cubiques}

On a vu dans l'exemple \ref{exempleder} que le groupe de Galois sur
un corps $K$ d'un polynôme de degré $2$ et soit il est égale a l'identité,
soit il est isomorphe au groupe de permutation $S_{2}\simeq\mathbb{Z}/2\mathbb{Z}$.\\
Dans cette section nous allons nous intéresser particulièrement aux
groupes de Galois des polynômes de degré $3$, pour cela, rappelons
que dans le corps $K$ que on a supposer de caractéristique différente
de $2$, un polynôme unitaire de degré $3$ s'écrit nécessairement
sous la forme $X^{3}+aX^{2}+bX+c$, où les coefficients $a,b,c$ sont
dans $K$, cependant on a vu dans le chapitre II que on peut toujours
ramener la forme précédente a un polynôme de la forme $X^{3}+pX+q$,
avec $p,q$, dans $K$, maintenant si on pose $p(X)=X^{3}+pX+q$,
avec $p(X)$ un polynôme dans $K[X]$, alors son discriminant comme
on la vu au chapitre II est donné par la formule explicite suivante
: $D_{p}=-(4p^{3}+27q^{2})$, dans la suite on s'intéresseras d'une
manière particulière aux résultats obtenus après le calcul de ce discriminant
et finalement on va déterminé d'une manière précise le groupe de Galois
associé, pour débuter considérons le théorème suivant : \begin{theorem}\label{theoreme3.1}
	Soit Soit $f(X)\in K[X]$ un polynôme irréductible et séparable de
	degré $n$.\\
	Le groupe de Galois de $G(f/K)$ est un sous groupe de $A_{n}\iff D_{f}$
	est un carré dans $K$. \end{theorem} 
\begin{proof}
	Soit $\Delta=\prod_{i<j}(\al_{j}-\al_{i})\ne0,\;1\leq i,j\leq n$,
	donc $\Delta\in K(\al_{1},\ldots,\al_{n})$ et $\Delta^{2}=D_{f}\in K$.
	par suite le discriminant de $f$ est dans $K$ si et seulement si
	$\Delta\in K$.\\
	Pour $\sigma\in Gal(K(\al_{1},\ldots,\al_{n}):K)$, soit $\epsilon_{\sigma}=\pm1$
	la signature de $\sigma$, de plus $\sigma$ est identifié soit a
	une permutation paire soit a une permutation impaire, on pose alors
	\[
	\sigma(\Delta)=\prod_{i<j}(\sigma(\al_{j})-\sigma(\al_{i}))=\epsilon_{\sigma}\prod_{i<j}(\al_{j}-\al_{i})=\epsilon_{\sigma}\Delta,
	\]
	alors $\sigma(\Delta)=\pm\Delta$, puisque $\Delta\ne0$, alors $\Delta\ne-\Delta$,
	on a donc$\sigma\in A_{n}\iff\epsilon_{\sigma}=1$, alors $\sigma\in A_{n}\iff\sigma(\Delta)=\Delta$,
	on en déduit que $G(f/K)\simeq A_{n}\iff\Delta\in K$ 
\end{proof}
\begin{corollary}\label{corollairechapi3} Soit $f(X)\in K[X]$ un
	polynôme irréductible et séparable de degré $3$.\\
	Si $D_{f}=\delta^{2},\,\delta\in K$, ( i.e. $D_{f}$ est un carré
	dans $K$ ), alors le groupe de Galois de $f(X)$ sur $K$, $G(f/K)\simeq A_{3}$,
	sinon si $D_{f}$ n'est pas un carré dans $K$, (i.e. $\nexists\delta\in K,\:\text{tel que}\,D_{f}=\delta^{2}$),
	alors $G(f/K)\simeq S_{3}$. \end{corollary} 
\begin{proof}
	D'après le théorème \ref{thoreme3}, comme $f(X)$ est irréductible
	alors $G(f/K)$ est isomorphes a un sous-groupe transitif de $S_{3}$,
	on sait que les deux seules groupes transitifs de $S_{3}$, Sont le
	groupe alterné $A_{3}$ et le groupe symétrique $S_{3}$, et d'après
	le théorème \ref{theoreme3.1}, $G(f/K)\simeq A_{3}\iff D_{f}$ est
	un carré, sinon on aura $G(f/K)\simeq S_{3}$ 
\end{proof}

\begin{theorem} \label{ThCh3} Soit $f(X)\in K[X]$ un polynôme séparable
	de degré $3$ et soit $\Delta=D_{f}$, si $\alpha$ est une racine
	de de $f(X)$, alors le corps de décomposition de $f(X)$ sur $K$
	est $Dec_{K}(f(X))=K(\al,\sqrt{\Delta})$, en particulier, si $f(X)$
	est réductible alors $Dec_{K}(f(X))=K(\sqrt{\Delta})$ \end{theorem} 
\begin{proof}
	Posons $f(X)\in K[X]$ un polynôme unitaire et séparable de degré
	$3$, soit $\al_{1},\al_{2},\al_{3}$ les racines de $f(X)$ dans
	une extension de $K$, on pose alors $f(X)=(X-\al_{1})g(X)$, donc
	$\al_{2}$ et $\al_{3}$ sont des racines de $g(X)$, en particulier
	$g(\al_{1})\ne0$, nécessairement $g(X)$ est un polynôme de degré
	$2$ sur $K(\al_{1})[X]$. On peut alors écrire l'égalité suivante
	\[
	K(\al_{1},\al_{2},\al_{3})=K(\al_{1})(\al_{2},\al_{3})=K(\al_{1})(\sqrt{D_{g}}).
	\]
	Puisque $f(X)$ est unitaire et divisible par $g(X)$ sur $K(\al_{1},\al_{2},\al_{3})$
	; \\
	Alors 
	\[
	D_{f}=g(\al_{1})^{2}D_{g}
	\]
	et donc $K(\al_{1},\sqrt{D_{g}})=K(\al_{1},\sqrt{D_{f}})=K(\al_{1},\sqrt{\Delta})$.\\
	Si $f(X)$ est réductible, en choisissant $\al_{1}\in K$ racine
	de $f(X)$. Alors $K(\al_{1},\sqrt{D_{g}})=K(\Delta)$\\
	Il important que le corps de base $K$ ne serait pas de caractéristique
	$2$. La preuve qu'on a utilisé l'impose, car nous avons utilisé la
	forme quadratique qui ne fonctionne pas en caractéristique $2$. \\
	Alors une question se pose, \ref{ThCh3} peut il être prouver avec
	différents arguments en $carK=2$? ; La réponse est non ; en en effet,
	le théorème telle qu'il est écrit, il est faux dans le cas où $carK=2$.
	Si nous prenons le contre exemple suivant: $K=F(u)$ avec $carF=2$
	et $u$ un nombre transcendant sur $F$, et le polynôme $f(X)=X^{3}+uX+u$
	qui est irréductible dans $K[X]$ avec $D_{f}=u^{2}$, alors $K(\al_{1},\sqrt{\Delta}))=K(\al_{1},\sqrt{D_{f}})=K(\al_{1},u)=K(\al_{1})$.
	On peut montrer que le degré du corps de décomposition de $f(X)$
	sur $K$ est égale à $6$ et non $3$, et par conséquent $K(\al_{1},\sqrt{\Delta})$
	n'est pas le corps de décomposition de $f(X)$ sur $K$.\\
	Preuve que le corps de décomposition de $f(X)$ est de degré $6$
	sur $K$. En effet comme $\al_{1}$ est racine du $f(X)$ alors $f(X)$
	s'écrit de la manière suivante 
	\[
	f(X)=(X-\al_{1})g(X)\,\text{dans}\,K(\al_{1},\sqrt{D_{g}}),
	\]
	et donc on a $D_{g}$ divise $D_{f}$ et donc $u^{2}$ dans $K=F(u)$,
	alors $\exists\delta\in K\,\text{tel que}\,\delta D_{g}=u^{2}\,\text{, et}\,\delta>1$,
	et comme $D_{g}\in K$ alors il s'écrit de la manière unique $D_{g}=a+bu+cu^{2}$
	et comme $D_{g}<D_{f}=u^{2}$ alors $D_{g}=a+bu$ et posons $\delta=a'+b'u$
	ainsi calculons $\delta D_{g}=aa'+(ab'+a'b)u+bb'u^{2}$ nous trouverons
	que $\delta=u$ et $D_{g}=u$, par suite, on aura 
	\[
	f(X)\in K(\al_{1},\sqrt{D_{g}})=K(\al_{1},\sqrt{u})=Dec_{K}(f)

\subsection{Groupes de Galois des extensions quartiques}
Supposons que le polynôme $p(X) = X^4 + aX^3 + bX^2 + cX + d, \: a,b,c,d \in K$, unitaire et irréductible dans $K[X]$, donc $D_p \ne 0$, notons par $(1 \leq \al_{i} \leq 4)$, les racines de $p(X)$ dans un corps de décomposition de ce dernier, soit alors l'égalité suivante : 
\begin{equation}
	X^4 + aX^3 + bX^2 + cX + d = (X - \al_{1})(X-\al_{2})(X- \al_{3})(X - \al_4).
\end{equation} 
On a vu que un groupe de Galois d'un polynôme $f(X)$ cubique irréductible et séparable dans un corps $K[X]$, est entièrement déterminé par son discriminant $D_f = \Delta$, si oui ou non ce dernier est un carré dans $K$; chose qui peut être pensé en terme de polynôme quadratique a savoir un polynôme de la forme $X^2 - \Delta$ ayant une racine dans $K$, on peut alors pensé qu'un polynôme quartique dépend du comportement d'un polynôme cubique associé, pour cela on essayeras de déterminer un polynôme cubique ayant des racines dans le corps de décompositions de $p(X) = X^4 + aX^3 + bX^2 + cX + d, \: a,b,c,d \in K$ sur $K[X]$ notons le $L$, sachant que ces groupes de Galois étant dans $S_4$, on chercheras donc un polynôme a $4$ variables qui sous les 24 permutation possible admettrais $3$ valeurs.  \\
Cependant on sait que le sous-groupe $V$ est normal dans $S_4$, ce dernier laisse fixe les éléments suivant dans $L$ : 
$$\beta_{1} =\al_{1}\al_{2} + \al_{3}\al_{4}, \; \beta_{2} = \al_{1} \al_{3} + \al_{2}\al_{4}, \; \beta_{3} = \al_{1}\al_{4} + \al_{2}\al_{3}.$$ 
Les $1 \leq \beta_{i} \leq 3 \in L$ engendrent un sous-corps de $L$ noté $K(\beta_{1},\beta_{2},\beta_{3}),$ et ce sont des racines d'un polynôme $\phi(X)$ défini par \begin{equation}\label{equachap3}
	\phi(X) = 	(X-\beta_{1})(X-\beta_{2})(X-\beta_{3}).
\end{equation}
\begin{definition}
	le polynôme $\phi(X)$ est appelé résolvant cubique du polynôme $p(X)$. \\ 
	On développent la formule \eqref{equachap3} on aura  \begin{equation}\label{equa2chap3}
		\phi(X) = X^3 - (\beta_{1} + \beta_{2} + \beta_{3})X^2 + (\beta_{1}\beta_{2} + \beta_{1}\beta_{3} + \beta_{2}\beta_{3})X - (\beta_{1}\beta_{2}\beta_{3}).
	\end{equation}
	Par calcule on obtient les résultats suivantes : 
	\begin{enumerate}
		\item $$(\beta_{1} + \beta_{2} + \beta_{3}) = -(\al_{1}\al_{2} + \al_{3}\al_{4} + \al_{1}\al_{3}+ \al_{2}\al_{4} + \al_{1}\al_{4} + \al_{2}\al_{3}) = -b.$$
		\item \begin{equation*}
			\begin{split}
				(\beta_{1}\beta_{2} + \beta_{1}\beta_{3} + \beta_{2}\beta_{3}) = \al_{1}^2\al_{2}\al_{3} + \al_{1}\al_{2}^2\al_{4}+\al_{2}\al_{3}\al_{4}^2+\al_{2}\al_{3}\al_{4}^2  + \al_{1}^2\al_{2}\al_{4} + \al_{1}\al_{2}^2\al_{3} + \al_{1}\al_{3}\al_{4}^2 + \\ \al_{2}\al_{3}^2\al_{4}+\al_{1}^2\al_{3}\al_{4}+\al_{1}\al_{2}\al_{3}^2+\al_{1}\al_{2}\al_{4}^2 + \al_{2}^2\al_{3}\al_{4} = ac - 4d
			\end{split}
		\end{equation*}
		\item	$$(\beta_{1}\beta_{2}\beta_{3}) = -(\al_{1}\al_{2}+\al_{3}\al_{4})(\al_{1}\al_{3}+\al_{2}\al_{4})(\al_{1}\al_{4}+\al_{2}\al_{3}) = -(a^2d + c^2 - 4bd).$$
	\end{enumerate}
	Chose qui ramène l'équation \eqref{equa2chap3} a la formule suivante : $$ \phi(X) = X^3 - bX^2 + (ac-4d)X - (a^2d + c^2 - 4bd).$$ 
\end{definition}
\begin{remark}
	\begin{itemize}
		\item Dans le cas ou $p(X)$ est unitaire, pour vérifié si $p(X)$ est irréductible il suffit de voir si $\phi(X)$ est irréductible. 
		\item Si $a = b = 0$, alors $p(X) = X^4 + cX +d \Longrightarrow X^3 - (4d)X - c^2.$
	\end{itemize}
	
\end{remark}

\end{theorem}
\begin{proof}
	Si $G(p/\mathbb{Q}) \simeq \mathbb{Z}/4\mathbb{Z}$, alors $[Dec_\mathbb{Q}(p(X))] = 4$, tout racine de $p(X)$ génère une extension de $\mathbb{Q}$ de degré $4$, donc le corps généré sur $K$ par une seule racine de $p(X)$ contient nécessairement les autres racines, cela dit si $p(X)$ admet une seule racine réelle  alors il admet $4$ autre racine réelles, donc le nombre de racine réelles de $p(X)$ et soit $0$ soit $4$. \\
	Maintenant si $p(X)$ n'admet pas de racine réelle alors forcément il admettrait des racines complexes deux a deux conjugués, alors $D_p$ est un carré de : 
	\begin{equation}\label{equapro}
		(z-\bar{z})(z-w)(z-\bar{w})(\bar{z}-w)(\bar{z}-\bar{w})(w-\bar{w}) = |z-w|^2|z-\bar{w}|^2(z-\bar{z})(w-\bar{w}).
	\end{equation}
	Les différences $z- \bar{z}$ et $w - \bar{w}$ sont purement imaginaires et non nulles, puisque $z$ et $w$ ne sont pas des nombres réels, donc nécessairement leurs produit 
	est réel et non nul, ainsi donc si on met au carré l'égalité \eqref{equapro}, on trouve que $D_p > 0$.\\
	On dernier si $p(X)$ admet $4$ racines réelles alors le produits des différences de ses racines est nécessairement un nombre réel no nul, donc $D_p > 0$.
\end{proof}
\begin{example}
	Soit le polynôme $p(X) = X^4 + 4X^2 -2 \in \mathbb{Q}$, il est irréductible car il est d'Eisenstein en $t = 2$, sont discriminant est $D_p = - 18432$, sa résolvante cubique associé est 
	$\phi(X) = X^3 -4X^2 +8X -32 = (X-4)(X^2+8),$ donc d'après le théorème \ref{thechap3deg4} $G(p/\mathbb{Q}) \simeq D_4 , \, ou \, \mathbb{Z}/4\mathbb{Z}$. Or d'après le théorème \ref{thechap3deg4.1} et comme on $D_p < 0$ alors  $G(p/\mathbb{Q}) \simeq D_4$.
\end{example}

	Puisque $\alpha_{1} + \alpha_{2}$ et $\alpha_{1}\alpha_{2}$ sont dans $L$ et $[L / K] = 2$, chacun deux engendre l'extension $L$ sur $K$ dans le cas ou il ne sont pas de $K$, et cela se produit pour au moins un des deux nombres.\\
	Premièrement on suppose que $G(p/K) \simeq \mathbb{Z}/4\mathbb{Z}$ alors $L = K(\sqrt{\Delta})$, donc $X^2 + aX + (b- \beta)$ et $X^2 - \beta X + d$, se décomposent complément sur $K(\sqrt{\Delta})$, puisque leurs racines sont dans $L$.\\

\end{proof}

			\end{tabular}
			\caption{ Quelques exemples pour illustrer le corollaire \ref{corollaire D_4 et Z/4Z}}
		\end{center}
	\end{table}
\end{proof}
\subsection{Distinction entre $\mathbb{Z}/4\mathbb{Z}$ et $D_4$ (Méthode Classique)} 
Dans la suite nous donnerons la méthode classique pour déterminer les groupes de Galois d'uu polynôme de degré $4$; pour cela considérons le théorème suivant : 
\begin{theorem} \label{thancien}
	Sous les hypothèses précédente, soit $p(X)$ un polynôme irréductible, on suppose que $D_p = \Delta \not\in K^2$ et $\phi(X)$ réductible dans $K[X]$ alors $G(p/K)$ est décrit comme suit : 
	\begin{enumerate}
		\item si $p(X)$ est irréductible sur $K(\sqrt{\Delta}) \Longrightarrow G(p/K) \simeq D_4$.
		\item si $p(X)$ est réductible sur $K(\sqrt{\Delta}) \Longrightarrow G(p/K) \simeq \mathbb{Z}/4\mathbb{Z}$.
	\end{enumerate}
\end{theorem} 
\begin{proof}
	Si $G(p/K) \simeq D_4$, on sait d'après le treillis précédent que $Dec_K(p(X))$ est $K(\al_1,\sqrt{\Delta})$, puisque $[K(\al_1,\sqrt{\Delta}) / K] = 8 \, ; [K(\al_1,\sqrt{\Delta}) / K(\sqrt{\Delta})] = 4$, alors $p(X)$ doit être nécessairement irréductible sur $K(\sqrt{\Delta})$. \\ 
	Si $G(p/K) \simeq \mathbb{Z}/4\mathbb{Z}$, $[Dec_K(p(X)) / K(\sqrt{\Delta})] = 2$, alors $p(X)$ est réductible sur  $K(\sqrt{\Delta})$.
\end{proof}
\begin{example}
	Posons $K=\mathbb{Q}$, les polynômes $p_1(X)=X^4+3X+2$ et $p_2(X)=X^2+5X+5$ dont les discriminants sont $D_{p_1}$ et $D_{p_2}$ présentés dans le tableaux suivants, vérifient tout les deux les hypothèses du théorème \ref{thancien}. Nous allons utiliser ce théorème afin de leur trouver leurs groupes de Galois sur $\mathbb{Q}$\\
	\begin{table}[!h]
		\begin{center}
			\begin{tabular}{|c||c||c||c|}
				\hline
				$p_i(X)$ & $D_{p_i}$ & $\phi_i(X)$ & $G(p_i/K)$\\
				\hline
				$p_1(X)$ & $21\times 15^2$ & $(X+3)(X^2-3X-3)$ & $D_4$\\
				$p_2(X)$ & $5 \times 55^2$ & $(X-5)(X^2+5X+5)$ & $\mathbb{Z}/4\mathbb{Z}$\\
				\hline
			\end{tabular}
			\caption{ Les groupes de Galois de $p_1$ et de $p_2$}
		\end{center}
	\end{table}\\
	Afin de trouver le groupe de Galois en utilisant le théorème \ref{thancien}, nous devons savoir si $p_1$ est irréductible sur $\mathbb{Q}(\sqrt{21})$ et si $p_2$ est irréductible sur $\mathbb{Q}(\sqrt{5})$. En fait, ce théorème dit que si un polynôme $P$ est irréductible sur un sous corps quadratique de son corps de décomposition, alors $G(P/K)\simeq D_4$ ; Si $P$ est réductible sur un sous-corps de son corps de décomposition, alors $G(P/K)\simeq \mathbb{Z}/2^2\mathbb{Z}$.\\
	Ces polynômes sont tous les deux irréductibles sur $\mathbb{Q}$, alors leurs racines sont de degré $4$ sur $\mathbb{Q}$ et par conséquent ils n'appartiennent pas au sous-corps quadratique du corps de décomposition de ces deux polynômes. Ceci veut dire que s'ils sont réductibles sur les sous-corps quadratiques de leurs corps de décompositions, ceci doit être un produit de deux polynômes de degré $2$.\\
	Afin de connaître si $p_1(X)$ est irréductible sur $\mathbb{Q}(\sqrt{21})$, nous supposons que $p_1$ se factorise de la manière suivante $$p_1(X)=X^4+3X+3=(X^2+AX+B)(X^2+CX+D)$$ et nous calculons les coefficients des polynômes de degré $2$. Nous trouverons le système suivant
	\begin{equation}
		\begin{cases}
			A+C=0\\
			B+D+AC=0\\
			AD+BC=3\\
			BD=3\\
		\end{cases}
	\end{equation} 
	ce qui implique que $C=-A$ et $D=-AC-B=A^2-B$, alors la troisième condition devient $A(A^2-2B)=3$. Ceci nécessite que $A\ne 0$ et par suite nous pourrons trouver $B$: $$B=\dfrac{A^3-3}{2A}$$
	Et par suite nous aurons la quatrième condition $BD=3$ devient $$3=\dfrac{A^3-3}{2A}\left( A^2-\dfrac{A^3-3}{2A}\right) =\dfrac{A^6-9}{4A^2}$$ et par suite $$0=A^6-12A^2-9=(A^2)(A^4-3A^2-3)$$ Cette dernière équations doit avoir une solution $A$ dans $\mathbb{Q}(\sqrt{21})$. L'équation $A^2+3=0$ n'a aucune solution dans $\mathbb{R}$ donc de même dans $\mathbb{Q}(\sqrt{21})$. Comme $X^4-3X^2-3$ est irréductible sur $\mathbb{Q}$, alors ses racines sont de degrés $4$ sur $\mathbb{Q}$ et celles-ci ne peuvent être dans $\mathbb{Q}(\sqrt{21})$, ce qui est absurde. Et ceci prouve bien que $p_1(X)$ est irréductible sur $\mathbb{Q}(\sqrt{21})$. Et ceci implique que $$G(p_1/\mathbb{Q})\simeq D_4$$ et ceci est exactement le résultat que nous avons vu dans le tableau de l'exemple \ref{EXEMPLED4}.\\
	Nous utilisons exactement la même méthode de factorisation qu'on a vu pour le polynôme $p_1(X)$ pour factoriser $p_2(X)$ sur $\mathbb{Q}(\sqrt{5})$, nous aurons à la fin ces égalités suivantes : $$0=A^6-20A^2-25=(A^2-5)(A^4+5A+5)$$ cette équation a pour $A=\sqrt{5}$ une solution dans $\mathbb{Q}(\sqrt{5})$, et ceci permet la factorisation sur ce corps $$p_2(X)=X^4+5X+5=\left( X^2+\sqrt{5}X+\dfrac{5-\sqrt{5}}{2}\right) \left( X^2-\sqrt{5}X+\dfrac{5+\sqrt{5}}{2}\right) $$ ce qui implique que $$G(p_2/\mathbb{Q})\simeq \mathbb{Z}/4\mathbb{Z}$$ 
\end{example}
